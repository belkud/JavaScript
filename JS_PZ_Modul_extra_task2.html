    {name:'eggs', price: 80, delivery:50},
    {name:'bread', price: 100, delivery:20},
    ]
    
    let infoAboutItems = document.getElementById('infoAboutItems')  as HTMLDivElement
    infoAboutItems.innerHTML = JSON.stringify(items)
1. Выведите на экран общую стоимость всех товаров вместе со стоимостью доставки. 
    
let priceWithDelivery = items.map(el=>el.price + el.delivery) //считаем построчно стоимость товара вместе с доставкой

let accum = 0 // сюда запишем общую стоимость
priceWithDelivery.forEach(el=>accum+=el) // считаем общую стоимость

infoAboutItems.innerHTML += `Общая стоимость всех товаров = 
${accum} рублей `      // выводим общую стоимость товаров в HTML файл
